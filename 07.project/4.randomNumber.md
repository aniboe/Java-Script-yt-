let randumNumber = parseInt(Math.random() * 100 + 1) // random number between 1 - 100

const submit = document.querySelector('#subt') // submit button

const userInput = document.querySelector('#guessField') // input box

const guessSlot = document.querySelector('.guesses') // Previous Guess

const remaining = document.querySelector('.lastResult') // Guesses Remaining

const lowOrHigh = document.querySelector('.lowOrHi') // empty (we have to show wether random number is high or low)

const startOver = document.querySelector('.resultParas') // dont know now


const p = document.createElement('p') // later used to make a button inside startOver

let prevGuesses = []

let numberGuess = 1 //guesses count to know when to stop 

let playGame = true



if (playGame){
  submit.addEventListener('click',function(evnt){
    evnt.preventDefault()
    const guess = parseInt(userInput.value)
    // console.log(guess)
    validateGues(guess)
  })
}



function validateGues(guess){
  // wether number is valid or not
  if (isNaN(guess)){
    alert('please enter a valid number')

  }else if(guess < 1){
    alert('please enter a number greater than 1')

  }else if(guess > 100){
    alert('please enter a number smaller than 100')

  }else{
    // prevGuesses.push(guess)

    // if (numberGuess >= 10){
    //   displayMessag('game over')
    //   console.log(`random number was${randumNumber}`)
    //   endGame()

    // }else{
    //   displayGuess(guess)
    //   checkGuess(guess)
    // }

    // problem with above code is that it doesnt show last entered value and stops game, so
    
    // we put following three line ouside of the if condition so they can be shown regardless
    prevGuesses.push(guess)
    displayGuess(guess)
    checkGuess(guess)

    numberGuess ++ // adding it here from line 103

    if (numberGuess > 10){
      displayMessag(`Game Over! The number was ${randumNumber}`)
      endGame()
    } // there is some correction in checkGuess()
  }
}



function checkGuess(guess){
  // wether guess is correct

  if(guess === randumNumber){
    displayMessag('you guessed it right')
    endGame()

  }else if ( guess < randumNumber){
    displayMessag(`random number is too smaller`)

  }else if ( guess > randumNumber){
    displayMessag(`random number is too greater`)
  }
}



function displayGuess(guess){
  // clear input
  userInput.value = ''

  // add guesses in inner html
  guessSlot.innerHTML += `${guess} ` 

  // numberGuess ++ // removing it from here
  remaining.innerHTML = `${10 - numberGuess}`

  // decease number of play left
}



function displayMessag(message){
  // is random number is low or high
  lowOrHigh.innerHTML = `<h2>${message}</h2>`

}

function endGame(){
  //
  playGame = false

  userInput.value = ''
  userInput.setAttribute('disabled', '')

  // creating start button
  p.classList.add('button')
  p.innerHTML = `<h2 id = 'newGame'> Start new game </h2>`
  startOver.appendChild(p)

  newGame()

}



function newGame(){
  // 
  const newGame = document.querySelector('#newGame')
  newGame.addEventListener('click', function(){

    randumNumber = parseInt(Math.random() * 100 + 1)
    let prevGuesses = []
    let numberGuess = 1
    guessSlot.innerHTML = ''
    remaining.innerHTML = `${10 - numberGuess}`

    userInput.removeAttribute('disabled', '')
    userInput.innerHTML = ''

    startOver.removeChild(p) // removing appended child

    playGame = true
  })
}
