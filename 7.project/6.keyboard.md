#### java script

```javascript
const insert = document.getElementById('insert')
window.addEventListener('keydown',(evnt)=>{
  insert.innerHTML = `
  <div class = 'color' >
    <table>
      <tr>
        <th>key</th>
        <th>keycode</th>
        <th>code</th>
      </tr>
      <tr>
        <td>${evnt.key === ' ' ? 'Space':evnt.key}</td>
        <td>${evnt.keyCode}</td>
        <td>${evnt.code}</td>
      </tr>
    </table>
  </div>
  `
})

```
